$2.8.1$解：类$For$的定义如下：
\begin{lstlisting}
class For extends Stmt{
  Expr E1, E2, E3;
  Stmt S;
  public For(Expr x1, Expr x2, Stmt y){
	E1 = x1; E2 = x2; E3 = x3; S = y;
	loop = newlabel(); end = newlabel();
  }
  public void gen(Expr x3){
	E1.gen();
	emit( loop + ":" );
	Expr n = E2.rvalue();
	emit( "ifFalse " + n.toString() 
	   + " goto " + end );
	S.gen();
	E3.gen();
	emit( "goto " + loop );
	emit( end + ":" );
  }
}
\end{lstlisting}
